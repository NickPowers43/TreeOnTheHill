#version 300 es

precision highp float;

uniform sampler2D positionTex;

uniform mat4 modelMat;
uniform mat4 viewMat;
uniform mat4 projMat;

layout(location = 0) in vec3 aVertexVelocity;

void main()
{
	vec3 positionTexCoord = vec3(float(gl_VertexID) * (1.0 / 1024.0), 0.5, 0.0);
	vec4 position = texture(positionTex, positionTexCoord.xy);

	gl_Position = projMat * viewMat * vec4(aVertexVelocity,1);
	gl_PointSize = 10.0;
}
